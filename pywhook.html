 13    """
 14    Lightweight Python client for interacting with https://webhook.site.
 15
 16    Provides:
 17
 18    - Token creation and deletion
 19    - Request polling and background callbacks
 20    - Default response configuration
 21    - Automatic cleanup via context manager
 22
 23    Example
 24    -------
 25    >>> token_data = Webhook.create_token()
 26    >>> with Webhook(token_data["uuid"]) as hook:
 27    ...     print(hook.url)
 28    ...     req = hook.wait_for_request(timeout=10)
 29    ...     print(req)

240    def wait_for_request(self, timeout: int = 15, interval: float = 0.1) -> Dict[str, Any]:
241        """
242        Block until a new request arrives or timeout is reached.
243
244        Parameters
245        ----------
246        timeout : int
247            Maximum seconds to wait.
248        interval : float
249            Polling interval in seconds.
250
251        Returns
252        -------
253        dict
254            The new request object.
255
256        Raises
257        ------
258        TimeoutError
259            If no new request arrives within the timeout.
260        """
261        latest = self.get_latest_request()
262        last_uuid = latest.get("uuid") if latest else None
263        start = time.time()
264
265        while time.time() - start < timeout:
266            req = self.get_latest_request()
267            if req and req.get("uuid") != last_uuid:
268                return req
269            time.sleep(interval)
270
271        raise TimeoutError(f"No new request after {timeout} seconds")

275    def on_request(self, callback: Callable[[Dict[str, Any]], None], interval: float = 0.1) -> None:
276        """
277        Register a callback that fires for every new request.
278
279        Parameters
280        ----------
281        callback : Callable[[dict], None]
282            Function called with the request dictionary.
283        interval : float
284            How often to poll for new requests (seconds).
285
286        Notes
287        -----
288        The callback runs in a background daemon thread. Transient API errors
289        are ignored, so your function may miss some requests if the API fails.
290        """
291        def listener(last_uuid: Optional[str]):
292            while not kill_event.is_set():
293                try:
294                    req = self.get_latest_request()
295                    if req and req.get("uuid") != last_uuid:
296                        callback(req)
297                        last_uuid = req.get("uuid")
298                except WebhookError:
299                    pass
300                time.sleep(interval)
301
302        kill_event = threading.Event()
303        latest = self.get_latest_request()
304        last_uuid = latest.get("uuid") if latest else None
305
306        thread = threading.Thread(target=listener, args=(last_uuid,), daemon=True)
307        self._on_request_threads.append({"thread": thread, "kill_event": kill_event})
308        thread.start()

311    def callbacks_on_request(self) -> List[Dict[str, Any]]:
312        """Return a list of active request listener threads and kill events."""

315    def detach_callback(self, index: int) -> List[Dict[str, Any]]:
316        """
317        Stop and remove a callback by its index.
318
319        Parameters
320        ----------
321        index : int
322            Index of the callback to remove.
323
324        Returns
325        -------
326        List[Dict[str, Any]]
327            Updated list of active callbacks.
328
329        Raises
330        ------
331        IndexError
332            If the index is out of range.
333        """
334        if index >= len(self._on_request_threads):
335            raise IndexError("Callback index out of range.")
336
337        entry = self._on_request_threads[index]
338        entry["kill_event"].set()
339        entry["thread"].join()
340        self._on_request_threads.pop(index)
341        return self.callbacks_on_request
342
343    def detach_all_callbacks(self) -> None:
344        """Stop and remove all active request callbacks."""
345        for entry in self._on_request_threads:
346            entry["kill_event"].set()
347            entry["thread"].join()
348        self._on_request_threads.clear()

385    def download_request_content(self, request: Dict[str, Any]) -> Dict[str, Dict[str, Any]]:
386        """
387        Download uploaded files attached to a request.
388
389        Parameters
390        ----------
391        request : dict
392            Request object returned by webhook.site API.
393
394        Returns
395        -------
396        dict[str, dict]
397            Mapping of file keys to their data including bytes, filename, and content type.
398
399        Raises
400        ------
401        WebhookError
402            If request object is invalid or download fails.
403        """
404        request_id = request.get("uuid") or request.get("id")
405        if not request_id:
406            raise WebhookError("Request object missing 'uuid' or 'id'.")
407
408        files = request.get("files", {})
409        if not files:
410            return {}
411
412        out = {}
413        for key, file in files.items():
414            url = f"{self.BASE_URL}/token/{self.token_id}/request/{request_id}/download/{file['id']}"
415            try:
416                response = requests.get(url)
417                response.raise_for_status()
418            except requests.RequestException as e:
419                raise WebhookError(f"Failed to download file {file['id']}: {e}") from e
420
421            out[key] = {
422                "id": file["id"],
423                "filename": file["filename"],
424                "name": file["name"],
425                "bytes": response.content,
426                "size": file["size"],
427                "content_type": file["content_type"],
428            }
429        return out
